Lemma MboxPendProof:
    forall vl p r,
      Some p =
      BuildPreA´ os_api OSMboxPend
                 mbox_pendapi vl ->
      Some r =
      BuildRetA´ os_api OSMboxPend
                 mbox_pendapi vl ->
      exists t d1 d2 s,
        os_api OSMboxPend = Some (t, d1, d2, s) /\
        {|OSQ_spec , GetHPrio , I, r, Afalse|}|- {{p}}s {{Afalse}}.
  Hint Resolve CltEnvMod.beq_refl.
  Require Import MboxPendPart2.
