Fixpoint convert_api_spec (ls : list (fid * osapi)) : fid -> option osapi :=
  match ls with
    | nil => fun _ : fid => None
    | (id, imp) :: ls´ =>
      fun id´ : fid =>
        if Zbool.Zeq_bool id id´ then Some imp
        else convert_api_spec ls´ id´
  end.

Fixpoint convert_int_spec (ls : list (hid * int_code)) : hid -> option int_code :=
