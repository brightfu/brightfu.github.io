Lemma encrit1_rule_sound : forall Spec sd I r ri isr is cs i aop,
                         RuleSem Spec sd I r ri
                (OS[isr, true, is, cs] ** (ATopis i) ** (Apure (i <= INUM)) //\\ <|aop|>)
                  (sprim encrit)
                (OS[isr, false, is, true::cs] ** (invlth_isr I O i)//\\ <|aop|>).
