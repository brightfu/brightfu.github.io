Lemma seq_rule_sound_aux : forall I r ri p´ q FSpec sd s2,
        (forall o O op, (o, O, op) |= p´ ->
        MethSimMod FSpec sd (nilcont s2) o op O I r ri (lift q))->
        forall c ke ks aop o O,
        MethSimMod FSpec sd (c, (ke, ks)) o aop O I r ri (lift p´) ->
        MethSimMod FSpec sd (c, (ke, (AddStmtToKs s2 ks))) o aop O I r ri (lift q) .

Lemma seq_rule_sound : forall Spec sd I r ri p p´ q s1 s2,
                            RuleSem Spec sd I r ri p s1 p´ ->
                            RuleSem Spec sd I r ri p´ s2 q ->
                            RuleSem Spec sd I r ri p (sseq s1 s2) q .
