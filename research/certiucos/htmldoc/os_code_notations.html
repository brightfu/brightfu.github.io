Definition cfun := (ident * function).
Fixpoint convert_cfuns (ls : list cfun) : fid -> option function :=
  match ls with
    | nil => fun _ => None
    | (x :: ls´)%list => match x with
                           | (id, imp) =>
                             fun (id´ : fid) =>
                               if Zbool.Zeq_bool id id´ then Some imp
                               else (convert_cfuns ls´) id´
                         end
  end.
